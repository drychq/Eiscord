\section{接口与数据设计}

\subsection{HTTP REST API}

所有接口以 \texttt{/api/v1} 为前缀，认证通过 Authorization: Bearer JWT。

\CommandSpec{RegisterUser}{
创建平台账号和基础用户档案。
}{
\texttt{username}、\texttt{email\_or\_phone}、\texttt{password}。
}{
注册结果、\texttt{user\_id}。
}{
仅访客用户可发起。
}{
用户名或邮箱重复；密码不合规。
}{
可触发 \texttt{NotificationCreated}。
}{
User、AuditLog。
}

\CommandSpec{LoginUser}{
验证登录凭证并创建有效会话。
}{
\texttt{login\_identifier}、\texttt{password}。
}{
会话标识、用户资料摘要、社区列表、好友摘要、未读与通知摘要。
}{
账号必须存在且可登录。
}{
密码错误；账号禁用。
}{
\texttt{PresenceChanged}、\texttt{UnreadUpdated}。
}{
User、AuthSession、Membership。
}

\CommandSpec{SendMessage}{
发送私聊或文本频道消息。
}{
会话/频道标识、文本内容、附件列表、提及列表、客户端临时消息标识。
}{
已持久化消息、分发结果。
}{
发送方有会话查看和发言权限。
}{
内容为空或超长；附件非法；无权发送；重复提交。
}{
\texttt{MessageCreated}、\texttt{UnreadUpdated}、\texttt{NotificationCreated}。
}{
Message、Attachment、ReadState、Notification。
}

\CommandSpec{LoadMessages}{
分页加载消息历史。
}{
会话/频道标识、游标、页大小。
}{
消息列表、下一页游标、未读位置。
}{
请求者有查看权限。
}{
无权查看；游标无效。
}{
可配合 \texttt{UnreadUpdated}。
}{
Message、ReadState。
}

\CommandSpec{CreateServer}{
创建社区及默认结构。
}{
社区名称、简介、图标附件。
}{
社区详情、默认角色、默认频道。
}{
用户已登录且未超创建限制。
}{
输入非法；数量超限。
}{
\texttt{MemberJoined}、\texttt{ChannelChanged}。
}{
Server、Membership、Role、Channel。
}

\CommandSpec{JoinServer}{
通过邀请加入社区。
}{
\texttt{invite\_code}。
}{
成员记录、默认角色、可见频道列表。
}{
用户已登录；邀请有效。
}{
邀请无效或过期；用户已加入。
}{
\texttt{MemberJoined}。
}{
Invitation、Membership、Role。
}

\CommandSpec{JoinVoiceChannel}{
加入语音频道。
}{
\texttt{channel\_id}、初始静音/闭麦状态。
}{
语音会话状态、频道成员列表。
}{
有查看和加入语音频道权限。
}{
频道不存在或非语音类型；无权加入。
}{
\texttt{VoiceMemberJoined}。
}{
VoiceSession、Channel、Membership。
}

\subsubsection{完整 API 端点列表}

\toprule
\textbf{模块} & \textbf{方法/路径} & \textbf{说明} \\
\midrule

\toprule
\textbf{模块} & \textbf{方法/路径} & \textbf{说明} \\
\midrule
\endhead
Auth & POST /auth/register, /login, /refresh, /logout & 注册/登录/刷新/登出 \\
Users & GET /users/me, PATCH /users/me/profile, /presence & 用户信息/资料/状态 \\
Friends & GET /friends, POST /friend-requests, /accept, /reject & 好友列表与请求 \\
        & GET /dm-conversations & DM 会话列表 \\
Servers & POST/GET /servers, /servers/:id & 创建/列表/详情 \\
        & POST /servers/join, /:id/leave & 加入/离开 \\
        & POST /servers/:id/roles, PATCH/DELETE /roles/:rid & 角色管理 \\
Channels & POST /servers/:sid/channels & 创建频道 \\
         & PATCH/DELETE /channels/:id & 更新/删除频道 \\
Messages & GET/POST /channels/:cid/messages & 频道消息加载/发送 \\
         & GET/POST /dm-conversations/:cid/messages & DM 消息加载/发送 \\
         & POST /read-states & 标记已读 \\
         & POST /messages/:mid/delete & 删除消息 \\
Voice & POST /voice/channels/:cid/join & 加入语音 \\
      & POST /voice/sessions/:sid/leave & 离开语音 \\
      & PATCH /voice/sessions/:sid/state & 更新状态 \\
Attachments & POST /attachments/init, /:id/complete & S3 上传 \\
Notifications & GET /notifications, POST /notifications/read & 列表/标记已读 \\
Permissions & POST /permissions/check & 权限检查 \\
Health & GET /health & 健康检查 \\

\end{longtable}
